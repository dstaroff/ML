coords = list()

coords_min = main(function=f,
                  x0=array([2]),
                  a=0,
                  b=4,
                  lambd=0.1,
                  epsilon=0.2,
                  t='constant')
print('{0}\n\t(X: {1}; Y: {2})'.format('constant', coords_min[0], f(coords_min, 1)))
coords.append((coords_min, 'constant'))

coords_min = main(function=f,
                  x0=array([2]),
                  a=0,
                  b=4,
                  lambd=0.1,
                  epsilon=0.2,
                  t='constant-descent')
print('{0}\n\t(X: {1}; Y: {2})'.format('constant-descent', coords_min[0], f(coords_min, 1)))
coords.append((coords_min, 'constant-descent'))

coords_min = main(function=f,

                  b=4,

                  epsilon=0.2,
                  t='optimal-descent')
print('{0}\n\t(X: {1}; Y: {2})'.format('optimal-descent', coords_min[0], f(coords_min, 1)))

coords = list()

                  x0=array([1, 2]),
                  a=0,
                  b=4,
                  lambd=0.1,
                  epsilon=200,
                  t='monte-carlo')
print('{0}\n\t(X: {1}; Y: {2}; Z: {3})'.format('monte-carlo', coords_min[0], coords_min[1], f1(coords_min, 2)))
